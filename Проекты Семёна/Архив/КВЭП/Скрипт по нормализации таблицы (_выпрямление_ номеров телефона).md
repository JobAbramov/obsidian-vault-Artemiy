function normalizeTable(rowEnd, rowStart = 1, last_out_row) {

let sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Оригинальная")

let sheet_out = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Нормализованная")

let qty_columns = 0;

  

for (let i = 1; ; i++) {

let curr_cell = sheet.getRange(1, i).getValue()

  

if (curr_cell === '') break;

  

qty_columns++;

}

  

//console.log(qty_columns)

let qty_rows = 0

let table_arr = []

  
  

for (let j = rowStart+1; j <= rowEnd+1; j++) {

let phone_cell = sheet.getRange(j, 8)

let phones = String(phone_cell.getValue()).split("\n")

  

phones.forEach((val, ind, arr) => {

let row_arr = []

for (let i = 1; i <= qty_columns; i++) {

if (i === 8) {

row_arr.push(val)

continue

}

if (i === 10) {

row_arr.push(GETLINK(j, i))

continue

}

row_arr.push(sheet.getRange(j, i).getValue())

}

qty_rows++;

table_arr.push(row_arr)

})

}

  
  

sheet_out.getRange(last_out_row+1, 1, qty_rows, qty_columns).setValues(table_arr)

//console.log(phones)

return qty_rows

  

}

  

function GETLINK(row, column) {

var sheet = SpreadsheetApp.getActiveSheet();

var range = sheet.getRange(row, column);

  

var value = range.getRichTextValue();

var url = value.getLinkUrl();

//console.log(value.getText())

return url;

}
